Build docker image
docker build -t omnistack:v01 .

Run command to access nodejs docker bin bash (Access to both projects)
docker run -ti -v $(pwd):/opt/ui omnistack:v01 /bin/bash

Run Server (NodeJs)
docker run -ti -v $(pwd)/backend:/opt/ui -p 81:3333 omnistack:v01 yarn dev

Run Server (React)
docker run -ti -v $(pwd)/web:/opt/ui -p 80:3000 omnistack:v01 yarn start

Run Server (React Native)
docker run -ti -v $(pwd)/mobile:/opt/ui -p 19002:19002 omnistack:v01 yarn start
